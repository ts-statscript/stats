Algorithms

Simple sort:

function mode_simple(x) {
  if (x.length === 0)
    return NaN;
  const counts = new Map;
  for (const val of x)
    counts.set(val, (counts.get(val) || 0) + 1);
  let max_count = 0, mode_val = NaN;
  for (const [val, count] of counts)
    if (count > max_count) {
      max_count = count;
      mode_val = val;
    }
  return mode_val;
}

Quick select:

function mode(x) {
  const len = x.length;
  if (len === 0)
    return;
  if (len === 1)
    return x[0];
  const counts = new Map;
  let maxCount = 0, modeValue = void 0;
  for (let i = 0;i < len; i++) {
    const value = x[i], count = (counts.get(value) || 0) + 1;
    counts.set(value, count);
    if (count > maxCount) {
      maxCount = count;
      modeValue = value;
    }
  }
  return modeValue;
}
